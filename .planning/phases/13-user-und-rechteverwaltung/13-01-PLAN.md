async linkKeycloakUser(teacherId: string, keycloakUserId: string) {
  const teacher = await this.prisma.teacher.findUnique({
    where: { id: teacherId }, include: { person: true }
  });
  if (!teacher) throw new NotFoundException('Teacher not found');
  return this.prisma.person.update({
    where: { id: teacher.personId },
    data: { keycloakUserId },
  });
}
async unlinkKeycloakUser(teacherId: string) {
  const teacher = await this.prisma.teacher.findUnique({
    where: { id: teacherId }, include: { person: true }
  });
  if (!teacher) throw new NotFoundException('Teacher not found');
  return this.prisma.person.update({
    where: { id: teacher.personId },
    data: { keycloakUserId: null },
  });
}

       export class EffectivePermissionsService {
         constructor(private prisma: PrismaService) {}
         async resolve(userId: string): Promise<EffectivePermissionRow[]> {
           const userRoles = await this.prisma.userRole.findMany({
             where: { userId }, include: { role: true },
           });
           const roleNames = userRoles.map(ur => ur.role.name);
           const rolePerms = await this.prisma.permission.findMany({
             where: { role: { name: { in: roleNames } } },
             include: { role: true },
           });
           const overrides = await this.prisma.permissionOverride.findMany({
             where: { userId },
           });
           const rows: EffectivePermissionRow[] = [
             ...rolePerms.map(p => ({
               action: p.action,
               subject: p.subject,
               granted: !p.inverted,
               conditions: p.conditions as Record<string, unknown> | null,
               interpolatedConditions: p.conditions
                 ? interpolateConditions(p.conditions as Record<string, unknown>, { id: userId })
                 : null,
               source: { kind: 'role' as const, roleName: p.role.name },
               reason: null,
             })),
             ...overrides.map(o => ({
               action: o.action,
               subject: o.subject,
               granted: o.granted,
               conditions: o.conditions as Record<string, unknown> | null,
               interpolatedConditions: o.conditions
                 ? interpolateConditions(o.conditions as Record<string, unknown>, { id: userId })
                 : null,
               source: { kind: 'override' as const },
               reason: o.reason ?? null,
             })),
           ];
           return rows;
         }
       }
       export type EffectivePermissionRow = {
         action: string;
         subject: string;
         granted: boolean;
         conditions: Record<string, unknown> | null;
         interpolatedConditions: Record<string, unknown> | null;
         source: { kind: 'role'; roleName: string } | { kind: 'override' };
         reason: string | null;
       };
