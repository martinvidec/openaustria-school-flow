       // use-update-user-roles.ts — example with Silent-4XX invariant
       export function useUpdateUserRoles(userId: string) {
         const qc = useQueryClient();
         return useMutation({
           mutationFn: (payload: { roleNames: string[] }) =>
             apiFetch(`/admin/users/${userId}/roles`, { method: 'PUT', body: JSON.stringify(payload) }),
           onSuccess: () => {
             toast.success('Rollen aktualisiert');
             qc.invalidateQueries({ queryKey: ['user-roles', userId] });
             qc.invalidateQueries({ queryKey: ['users'] });
             qc.invalidateQueries({ queryKey: ['effective-permissions', userId] });
           },
           onError: (err: any) => {
             // Silent-4XX invariant: every mutation has explicit onError
             if (err?.status === 409 && err?.problem?.type === 'schoolflow://errors/last-admin-guard') {
               // LastAdminGuardDialog handles this via error propagation; don't double-toast
               return;

               description: err?.problem?.detail ?? `Der Server hat die Anfrage abgelehnt (Status ${err?.status ?? 'unknown'}).`,
             });
           },
         });
       }
