export interface DataListColumn<T> {
  key: string;
  header: React.ReactNode;
  className?: string;
  cell: (row: T) => React.ReactNode;

  rows: T[];
  columns: DataListColumn<T>[];
  getRowId: (row: T) => string | number;

}
