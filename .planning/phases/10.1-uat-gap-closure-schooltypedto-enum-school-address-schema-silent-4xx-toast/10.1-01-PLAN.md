Canonical useMutation pattern (from useUpdateSchool):
```typescript
useMutation({
  mutationFn: async (input): Promise<Output> => {
    const res = await apiFetch(url, { method: 'PUT', body: JSON.stringify(input) });
    if (!res.ok) throw new Error('<user-facing German message>');
    return res.json();

  onSuccess: (server) => {
    qc.invalidateQueries({ queryKey: ... });

  },
  onError: (e: Error) => toast.error(e.message),     // ERROR TOAST ONLY HERE
});
```

```typescript
/* @vitest-environment jsdom */
import { renderHook, waitFor } from '@testing-library/react';

import { vi, describe, it, expect, beforeEach } from 'vitest';
import React from 'react';
import { toast } from 'sonner';

function wrapper({ children }: { children: React.ReactNode }) {

}

describe('useUpdateSchool silent-4xx guard', () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  it('does NOT call toast.success when backend returns 4xx', async () => {

    expect(toast.success).not.toHaveBeenCalled();
    expect(toast.error).toHaveBeenCalledTimes(1);
  });
});
```
