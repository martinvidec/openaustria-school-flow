     constructor(

     async findByClass(classId: string) {

     async applyStundentafel(classId: string, schoolType: string) {
       const cls = await this.prisma.schoolClass.findUnique({ where: { id: classId } });
       if (!cls) throw new NotFoundException('Klasse nicht gefunden');

       const existing = await this.prisma.classSubject.count({ where: { classId } });
       if (existing > 0) {
         throw new ConflictException({
           type: 'https://schoolflow.dev/errors/stundentafel-exists',
           title: 'Stundentafel bereits vorhanden',
           status: 409,
           detail: 'Setzen Sie die Stundentafel erst zurück, bevor Sie eine neue Vorlage übernehmen.',
         });
       }
       const rows = this.templates.getForYearLevel(schoolType, cls.yearLevel);
       // rows shape: [{ subjectShortName, weeklyHours, preferDoublePeriod? }, ...]
       // Resolve subjectShortName → subjectId (match within this schoolId)
       const subjects = await this.prisma.subject.findMany({ where: { schoolId: cls.schoolId } });
       const byShort = new Map(subjects.map((s) => [s.shortName, s.id]));
       return this.prisma.$transaction(
         rows.filter((r) => byShort.has(r.subjectShortName)).map((r) =>
           this.prisma.classSubject.create({
             data: {
               classId,
               subjectId: byShort.get(r.subjectShortName)!,
               weeklyHours: r.weeklyHours,
               isCustomized: false,
               preferDoublePeriod: r.preferDoublePeriod ?? false,
             },
           }),
         ),
       );
     }

     async updateClassSubjects(classId: string, dto: UpdateClassSubjectsDto) {
       const cls = await this.prisma.schoolClass.findUnique({ where: { id: classId } });
       if (!cls) throw new NotFoundException('Klasse nicht gefunden');
       const template = this.templates.getForYearLevel(cls.schoolId /* placeholder */, cls.yearLevel); // resolve using school.type

       return this.prisma.$transaction(async (tx) => {
         // Load existing rows
         const existing = await tx.classSubject.findMany({ where: { classId } });
         const existingByKey = new Map(existing.map((cs) => [`${cs.subjectId}`, cs]));
         const incomingByKey = new Map(dto.rows.map((r) => [`${r.subjectId}`, r]));

         // DELETE rows no longer present
         const toDelete = existing.filter((cs) => !incomingByKey.has(cs.subjectId));
         if (toDelete.length > 0) {
           await tx.classSubject.deleteMany({ where: { id: { in: toDelete.map((d) => d.id) } } });
         }

         // UPSERT incoming rows (update existing, create new). isCustomized flips true when weeklyHours differs from template default.
         const templateByShortOrSubject = new Map<string, number>(
           template.map((t: any) => [t.subjectShortName, t.weeklyHours]),
         );
         const subjects = await tx.subject.findMany({ where: { schoolId: cls.schoolId } });
         const subjByIdShort = new Map(subjects.map((s) => [s.id, s.shortName]));

         for (const row of dto.rows) {
           const short = subjByIdShort.get(row.subjectId) ?? '';
           const templateHours = templateByShortOrSubject.get(short);
           const isCustomized = templateHours === undefined ? true : templateHours !== row.weeklyHours;
           const existingRow = existingByKey.get(row.subjectId);
           if (existingRow) {
             await tx.classSubject.update({
               where: { id: existingRow.id },
               data: { weeklyHours: row.weeklyHours, isCustomized, preferDoublePeriod: row.preferDoublePeriod ?? existingRow.preferDoublePeriod },
             });
           } else {
             await tx.classSubject.create({
               data: { classId, subjectId: row.subjectId, weeklyHours: row.weeklyHours, isCustomized, preferDoublePeriod: row.preferDoublePeriod ?? false },
             });
           }
         }

         return tx.classSubject.findMany({ where: { classId }, include: { subject: true } });
       });
     }

     async resetStundentafel(classId: string, schoolType: string) {
       return this.prisma.$transaction(async (tx) => {
         await tx.classSubject.deleteMany({ where: { classId } });

         for (const r of rows) {

           if (!subjectId) continue;
           await tx.classSubject.create({
             data: { classId, subjectId, weeklyHours: r.weeklyHours, isCustomized: false, preferDoublePeriod: r.preferDoublePeriod ?? false },
           });
         }
         return tx.classSubject.findMany({ where: { classId }, include: { subject: true } });
       });
     }
