   import { test, expect } from '@playwright/test';

   const PREFIX = 'E2E-TEA-';
   test.describe('Admin Teachers CRUD — Desktop Happy Path', () => {
     test.beforeEach(async ({ page }) => { await loginAsAdmin(page); });

       // DELETE teachers by firstName prefix via GET + bulk DELETE
       const res = await request.get('/api/v1/teachers?schoolId=' + SCHOOL_ID, { headers: { Authorization: `Bearer ${token}` } });

       for (const t of body.data ?? []) {

           await request.delete(`/api/v1/teachers/${t.id}`, { headers: { Authorization: `Bearer ${token}` } });
         }
       }
     });

     test('TEACHER-CRUD-01 Admin creates a Lehrperson via TeacherCreateDialog', async ({ page }) => {
       await page.goto('/admin/teachers');
       await page.getByRole('button', { name: 'Lehrperson anlegen' }).first().click();
       await expect(page.getByRole('dialog')).toBeVisible();
       const vorname = `${PREFIX}${Date.now()}`;
       await page.getByLabel(/Vorname/).fill(vorname);
       await page.getByLabel(/Nachname/).fill('Huber');
       await page.getByLabel(/E-Mail/).fill(`${vorname.toLowerCase()}@schule.at`);
       await page.getByRole('button', { name: 'Lehrperson anlegen' }).last().click();

       await expect(page.getByText(vorname)).toBeVisible();

       const created = await request.post('/api/v1/teachers', {
         headers: { Authorization: `Bearer ${token}` },
         data: { schoolId: SCHOOL_ID, firstName: vorname, lastName: 'Muster', email: 'edit@t.at' },

       await page.goto(`/admin/teachers/${id}?tab=stammdaten`);
