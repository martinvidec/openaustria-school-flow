async function cleanupE2EImports(request: APIRequestContext): Promise<void> {
  const token = await getAdminToken(request);
  const list = await request.get(`${API}/imports`, { headers: { Authorization: `Bearer ${token}` } });
  if (!list.ok()) return;
  const jobs = (await list.json()) as Array<{ id: string; fileName?: string; originalFileName?: string }>;
  // Cleanup by filename match — these specs upload untis-happy/csv-happy/csv-malformed
  const e2eFiles = new Set(['untis-happy.xml', 'csv-happy.csv', 'csv-malformed.csv']);
  await Promise.all(
    jobs
      .filter((j) => e2eFiles.has(j.fileName ?? '') || e2eFiles.has(j.originalFileName ?? ''))
      .map((j) => request.delete(`${API}/imports/${j.id}`, {
        headers: { Authorization: `Bearer ${token}` },
      })),
  );
}
