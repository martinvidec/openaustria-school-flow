    - create(dto): prisma.subject.create

    - remove(id): Delete subject (cascades classSubjects, teacherSubjects)
    - addToClass(subjectId, classId, weeklyHours, groupId?): Create ClassSubject with isCustomized=true
    - removeFromClass(subjectId, classId): Delete ClassSubject
    - updateClassHours(classSubjectId, weeklyHours): Update weeklyHours and set isCustomized=true
    - getClassSubjects(classId): List all ClassSubject for a class with subject info
