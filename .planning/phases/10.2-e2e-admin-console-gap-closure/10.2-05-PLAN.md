```yaml
name: Playwright E2E

on:
  pull_request:
    branches: [main]
  push:
    branches: [main]

jobs:
  e2e:
    runs-on: ubuntu-latest
    timeout-minutes: 30
    services:
      postgres:
        image: postgres:17
        env:
          POSTGRES_USER: schoolflow
          POSTGRES_PASSWORD: schoolflow_dev
          POSTGRES_DB: schoolflow
        ports: ['5432:5432']
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
      redis:
        image: redis:7
        ports: ['6379:6379']
    env:
      DATABASE_URL: postgresql://schoolflow:schoolflow_dev@localhost:5432/schoolflow
      REDIS_URL: redis://localhost:6379
      KEYCLOAK_URL: http://localhost:8080
      KEYCLOAK_REALM: schoolflow
      KEYCLOAK_CLIENT_ID: schoolflow-api
      VAPID_PUBLIC_KEY: ${{ secrets.VAPID_PUBLIC_KEY || 'BPeB9cWlM3JmH9qOHVmDaEPUu-ge0KsRf-FV3e_Sw_l8LPw37q8MZ-ywlBp40jrAH_tlZJ8fbpaya27h6w3DgSE' }}
      VAPID_PRIVATE_KEY: ${{ secrets.VAPID_PRIVATE_KEY || '9zvNGSCLHASLT6X9uCUHuKWzj9IQUuGRiBhMXWArGHE' }}
      VAPID_SUBJECT: mailto:admin@schoolflow.example
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v4
        with: { version: 10 }
      - uses: actions/setup-node@v4
        with: { node-version: '24', cache: 'pnpm' }
      - run: pnpm install --frozen-lockfile

      - name: Start Keycloak (Docker Compose)
        run: docker compose -f docker/docker-compose.yml up -d keycloak keycloak-db
      - name: Wait for Keycloak
        run: timeout 120 bash -c 'until curl -sf http://localhost:8080/realms/schoolflow/.well-known/openid-configuration; do sleep 2; done'

      - name: Build shared + post-process ESM extensions
        run: |
          pnpm --filter @schoolflow/shared build
          cd packages/shared/dist && find . -name "*.js" -exec perl -i -pe "s{from '(\\.\\.?/[^']+?)'}{from '\$1.js'}g unless /\\.js'/; s{from \"(\\.\\.?/[^\"]+?)\"}{from \"\$1.js\"}g unless /\\.js\"/" {} + ; cd ../../..

      - name: Prisma migrate + seed
        run: |
          pnpm --filter @schoolflow/api exec prisma migrate deploy
          pnpm --filter @schoolflow/api exec prisma generate
          pnpm --filter @schoolflow/api exec tsx prisma/seed.ts

      - name: Build API + Web
        run: |
          pnpm --filter @schoolflow/api build
          pnpm --filter @schoolflow/web build

      - name: Install Playwright browsers
        run: pnpm --filter @schoolflow/web exec playwright install chromium --with-deps

      - name: Start services
        run: |
          nohup node apps/api/dist/main.js > api.log 2>&1 &
          nohup pnpm --filter @schoolflow/web preview --host 0.0.0.0 --port 5173 > web.log 2>&1 &
          timeout 60 bash -c 'until curl -sf http://localhost:3000/api/v1/health; do sleep 2; done'
          timeout 60 bash -c 'until curl -sf http://localhost:5173; do sleep 2; done'

      - name: Run Playwright
        run: pnpm --filter @schoolflow/web exec playwright test
        env:
          PLAYWRIGHT_BASE_URL: http://localhost:5173
          E2E_ADMIN_USER: admin-user
          E2E_ADMIN_PASS: admin123

      - name: Upload Playwright artifacts
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: playwright-traces
          path: |
            apps/web/test-results/
            apps/web/playwright-report/
            .planning/phases/10-schulstammdaten-zeitraster/uat-screenshots/
```

Decisions reflected above:
- Use Docker Compose only for Keycloak (the one that needs realm import). Postgres + Redis are GitHub services (faster, healthcheck-enabled).
- Seed uses `tsx prisma/seed.ts` (matches local startup memory).
- API built + run as `node dist/main.js` after shared-dist `.js`-extension post-process (per feedback memory).
- Web served via `vite preview` (not `dev`) for CI reproducibility and lower memory.

**If the workflow run reveals missing steps (e.g., Keycloak realm import race, missing env var), fix the workflow file iteratively and commit each fix as its own commit `ci(10.2-05): <what>`. Do NOT merge a red workflow — this is the UAT-ban gate.**
