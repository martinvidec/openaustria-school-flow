1. Create apps/web/src/hooks/useHomework.ts:
   - import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
   - import { apiFetch } from '../lib/api';
   - import type { HomeworkDto, CreateHomeworkRequest } from '@schoolflow/shared';

   - export function useHomework(schoolId: string, classSubjectId?: string):
     useQuery with queryKey ['homework', schoolId, classSubjectId]. queryFn: apiFetch(`/api/v1/schools/${schoolId}/homework${classSubjectId ? `?classSubjectId=${classSubjectId}` : ''}`). Returns HomeworkDto[].

   - export function useHomeworkByLesson(schoolId: string, classBookEntryId: string):
     useQuery with queryKey ['homework', 'by-lesson', schoolId, classBookEntryId]. queryFn: apiFetch(`/api/v1/schools/${schoolId}/homework/by-lesson/${classBookEntryId}`).

   - export function useCreateHomework(schoolId: string):
     useMutation. mutationFn: (dto: CreateHomeworkRequest) => apiFetch(`/api/v1/schools/${schoolId}/homework`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(dto) }). onSuccess: invalidate ['homework', schoolId] queries. Import toast from sonner, show success toast "Hausaufgabe erstellt".

   - export function useUpdateHomework(schoolId: string):
     useMutation. mutationFn: ({ id, ...dto }) => apiFetch PUT. onSuccess: invalidate queries.

   - export function useDeleteHomework(schoolId: string):
     useMutation. mutationFn: (id: string) => apiFetch DELETE. onSuccess: invalidate queries, toast "Hausaufgabe geloescht".

2. Create apps/web/src/hooks/useExams.ts:
   - Same pattern as homework hooks.

   - export function useExams(schoolId: string, classId?: string):
     useQuery with queryKey ['exams', schoolId, classId]. queryFn: apiFetch GET /exams?classId=.

   - export function useExamCollisionCheck(schoolId: string, classId: string, date: string, excludeId?: string):
     useQuery with queryKey ['exam-collision', schoolId, classId, date]. queryFn: apiFetch GET /exams/collision-check?classId=&date=&excludeId=. enabled: Boolean(classId && date). Returns ExamCollisionDto.

   - export function useCreateExam(schoolId: string):
     useMutation. Returns { exam, collision } from response. If collision.hasCollision, does NOT auto-toast (let dialog handle warning display). Otherwise toast "Pruefung eingetragen".

   - export function useUpdateExam(schoolId: string):
     useMutation with invalidation.

   - export function useDeleteExam(schoolId: string):
     useMutation with invalidation and toast.
  </action>
