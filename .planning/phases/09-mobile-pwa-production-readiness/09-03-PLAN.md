Build the Web Push notification backend: Prisma model, PushModule with CRUD + VAPID delivery, BullMQ processor for batched push, and extend NotificationService to trigger push alongside Socket.IO.

    **Step 1: Create PushProcessor (BullMQ)**
    At `apps/api/src/modules/push/push.processor.ts`:
    ```typescript
    @Processor(PUSH_QUEUE)
    export class PushProcessor extends WorkerHost {
      constructor(private readonly pushService: PushService) { super(); }

      async process(job: Job<{ userId: string; payload: { title: string; body: string; url?: string; tag?: string } }>): Promise<void> {
        try {
          await this.pushService.sendToUser(job.data.userId, job.data.payload);
        } catch (error) {
          // Log but do not rethrow -- push failure should not retry indefinitely
          this.logger.error(`Push delivery failed for user ${job.data.userId}`, error);
        }
      }
    }
    ```
