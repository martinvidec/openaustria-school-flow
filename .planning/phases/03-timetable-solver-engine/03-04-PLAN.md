5. Add constraints for custom constraint template types in `TimetableConstraintProvider.java`:
   - **NO_LESSONS_AFTER** (hard constraint — class must not have lessons after maxPeriod):
     ```java
     Constraint classTimeslotRestriction(ConstraintFactory factory) {
         return factory.forEach(Lesson.class)
             .join(ClassTimeslotRestriction.class,
                 Joiners.equal(Lesson::getClassId, ClassTimeslotRestriction::getClassId))
             .filter((lesson, restriction) ->
                 lesson.getTimeslot() != null &&
                 lesson.getTimeslot().getPeriodNumber() > restriction.getMaxPeriod())
             .penalize(HardSoftScore.ONE_HARD)
             .asConstraint("Class timeslot restriction");
     }
     ```
   - **SUBJECT_MORNING** (soft constraint — subject should be before latestPeriod):
     ```java
     Constraint subjectTimePreference(ConstraintFactory factory) {
         return factory.forEach(Lesson.class)
             .join(SubjectTimePreference.class,
                 Joiners.equal(Lesson::getSubjectId, SubjectTimePreference::getSubjectId))
             .filter((lesson, pref) ->
                 lesson.getTimeslot() != null &&
                 lesson.getTimeslot().getPeriodNumber() > pref.getLatestPeriod())
             .penalizeConfigurable()
             .asConstraint("Subject time preference");
     }
     ```
   - Register both constraints in the `defineConstraints()` return array.
   - Add "Subject time preference" to the default weights (value: 3).

```typescript
import { IsString, IsEnum, IsObject, IsOptional, IsBoolean } from 'class-validator';

export enum ConstraintTemplateType {
  BLOCK_TIMESLOT = 'BLOCK_TIMESLOT',         // Block specific time slots for a teacher
  SUBJECT_MORNING = 'SUBJECT_MORNING',       // Subject must be in morning periods
  NO_LESSONS_AFTER = 'NO_LESSONS_AFTER',     // No lessons after period N for a class
  SUBJECT_PREFERRED_SLOT = 'SUBJECT_PREFERRED_SLOT', // Subject preferred in specific slot
}

export class CreateConstraintTemplateDto {
  @IsEnum(ConstraintTemplateType)
  templateType!: ConstraintTemplateType;

  @IsObject()
  params!: Record<string, unknown>;
  // BLOCK_TIMESLOT: { teacherId: string, dayOfWeek: string, periods: number[] }
  // SUBJECT_MORNING: { subjectId: string, maxPeriod: number }
  // NO_LESSONS_AFTER: { classId: string, maxPeriod: number }
  // SUBJECT_PREFERRED_SLOT: { subjectId: string, dayOfWeek: string, period: number }

  @IsOptional()
  @IsBoolean()
  isActive?: boolean;
}

export class UpdateConstraintTemplateDto {
  @IsOptional()
  @IsObject()
  params?: Record<string, unknown>;

  @IsOptional()
  @IsBoolean()
  isActive?: boolean;
}

export class ConstraintTemplateResponseDto {
  id!: string;
  schoolId!: string;
  templateType!: string;
  params!: Record<string, unknown>;
  isActive!: boolean;
  createdAt!: Date;
}
```
