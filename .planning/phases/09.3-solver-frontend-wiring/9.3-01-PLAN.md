       let solverSocket: Socket | null = null;

       export function createSolverSocket(jwt: string): Socket {
         if (solverSocket?.connected) {
           return solverSocket;
         }
         const url = `${getSocketBaseUrl()}/solver`;
         solverSocket = io(url, {
           auth: { token: jwt },
           transports: ['websocket', 'polling'],
         });
         return solverSocket;
       }

       export function disconnectSolverSocket(): void {
         if (solverSocket) {
           solverSocket.disconnect();
           solverSocket = null;

       }

    2. **Create useSolverSocket hook** at apps/web/src/hooks/useSolverSocket.ts:
       ```typescript
       import { useEffect, useRef, useState } from 'react';
       import { useQueryClient } from '@tanstack/react-query';
       import { toast } from 'sonner';
       import type { Socket } from 'socket.io-client';
       import { createSolverSocket, disconnectSolverSocket } from '@/lib/socket';

       export interface SolveProgressEvent {
         runId: string;
         percent: number;
         hardScore: number;
         softScore: number;
       }

       export interface SolveCompleteEvent {
         runId: string;
         hardScore: number;
         softScore: number;
         lessonsCount: number;
       }

       export function useSolverSocket(jwt: string | null) {
         const queryClient = useQueryClient();
         const socketRef = useRef<Socket | null>(null);
         const [isConnected, setIsConnected] = useState(false);
         const [progress, setProgress] = useState<SolveProgressEvent | null>(null);
         const [lastResult, setLastResult] = useState<SolveCompleteEvent | null>(null);

         useEffect(() => {
           if (!jwt) return;
           const socket = createSolverSocket(jwt);
           socketRef.current = socket;

           socket.on('connect', () => setIsConnected(true));
           socket.on('disconnect', () => setIsConnected(false));

           socket.on('solve:progress', (event: SolveProgressEvent) => {
             setProgress(event);
           });

           socket.on('solve:complete', (event: SolveCompleteEvent) => {
             setLastResult(event);
             setProgress(null);
             queryClient.invalidateQueries({ queryKey: ['timetable'] });
             queryClient.invalidateQueries({ queryKey: ['timetable-runs'] });
             toast.success('Stundenplan-Generierung abgeschlossen', {
               description: `Hard: ${event.hardScore}, Soft: ${event.softScore}, Lessons: ${event.lessonsCount}`,
             });
           });

           return () => {
             disconnectSolverSocket();
             socketRef.current = null;
           };
         }, [jwt, queryClient]);

         return { isConnected, progress, lastResult };
       }
       ```

    1. **Create admin solver page** at `apps/web/src/routes/_authenticated/admin/solver.tsx`:
       ```tsx
       import { createFileRoute } from '@tanstack/react-router';
       import { useState } from 'react';
       import { useSchoolContext } from '@/stores/school-context-store';
       import { useAuth } from '@/lib/auth';
       import { useSolverSocket } from '@/hooks/useSolverSocket';
       import { Button } from '@/components/ui/button';
       import { Card } from '@/components/ui/card';
       import { apiFetch } from '@/lib/api';
       import { toast } from 'sonner';

       export const Route = createFileRoute('/_authenticated/admin/solver')({
         component: AdminSolverPage,
       });

       function AdminSolverPage() {
         const { jwt } = useAuth();
         const schoolId = useSchoolContext((s) => s.schoolId);
         const { isConnected, progress, lastResult } = useSolverSocket(jwt);
         const [isStarting, setIsStarting] = useState(false);

         const handleGenerate = async () => {
           if (!schoolId) return;
           setIsStarting(true);
           try {
             const res = await apiFetch(`/api/v1/schools/${schoolId}/timetable/solve`, {
               method: 'POST',
             });
             if (!res.ok) throw new Error('Failed to start solve');
             toast.info('Stundenplan-Generierung gestartet');
           } catch (error) {
             toast.error('Fehler beim Starten der Generierung');
           } finally {
             setIsStarting(false);
           }
         };

         return (
           <div className="max-w-[800px] mx-auto p-6 space-y-6">
             <div>
               <h1 className="text-[28px] font-semibold">Stundenplan-Generator</h1>
               <p className="text-muted-foreground mt-2">
                 Generieren Sie automatisch einen optimalen Stundenplan basierend auf den definierten Constraints.
               </p>
             </div>

             <Card className="p-6 space-y-4">
               <div className="flex items-center justify-between">
                 <div>
                   <h2 className="text-[20px] font-semibold">Status</h2>
                   <p className="text-sm text-muted-foreground">
                     {isConnected ? 'Verbunden' : 'Nicht verbunden'}
                   </p>
                 </div>
                 <Button
                   onClick={handleGenerate}
                   disabled={!schoolId || isStarting || progress !== null}
                 >
                   {isStarting ? 'Wird gestartet...' : 'Stundenplan generieren'}
                 </Button>
               </div>

               {progress && (
                 <div className="space-y-2">
                   <div className="flex justify-between text-sm">
                     <span>Fortschritt</span>
                     <span>{progress.percent}%</span>
                   </div>
                   <div className="w-full bg-muted rounded-full h-2">
                     <div
                       className="bg-primary h-2 rounded-full transition-all"
                       style={{ width: `${progress.percent}%` }}
                     />
                   </div>
                   <p className="text-xs text-muted-foreground">
                     Hard: {progress.hardScore} | Soft: {progress.softScore}
                   </p>
                 </div>
               )}
             </Card>

             {lastResult && (
               <Card className="p-6">
                 <h2 className="text-[20px] font-semibold mb-2">Letztes Ergebnis</h2>
                 <dl className="space-y-1 text-sm">
                   <div className="flex justify-between">
                     <dt>Run ID</dt>
                     <dd className="font-mono text-xs">{lastResult.runId}</dd>
                   </div>
                   <div className="flex justify-between">
                     <dt>Hard Score</dt>
                     <dd>{lastResult.hardScore}</dd>
                   </div>
                   <div className="flex justify-between">
                     <dt>Soft Score</dt>
                     <dd>{lastResult.softScore}</dd>
                   </div>
                   <div className="flex justify-between">
                     <dt>Lessons</dt>
                     <dd>{lastResult.lessonsCount}</dd>
                   </div>
                 </dl>
               </Card>
             )}
           </div>
         );
       }
       ```
