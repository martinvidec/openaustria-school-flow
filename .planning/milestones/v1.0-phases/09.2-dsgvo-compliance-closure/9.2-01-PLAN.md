    3. **kommunikation (1yr):** Use a transaction to delete MessageRecipient first (FK), then Message:
       ```typescript
       if (category === 'kommunikation') {
         const cutoff = new Date(Date.now() - retentionDays * 24 * 60 * 60 * 1000);
         const result = await this.prisma.$transaction(async (tx) => {
           const msgIds = await tx.message.findMany({
             where: { createdAt: { lt: cutoff } },
             select: { id: true },
           });
           const ids = msgIds.map((m) => m.id);
           if (ids.length === 0) return { count: 0 };
           await tx.messageRecipient.deleteMany({ where: { messageId: { in: ids } } });
           const deleted = await tx.message.deleteMany({ where: { id: { in: ids } } });
           return deleted;
         });
         this.logger.log(`Deleted ${result.count} expired message rows (retention: ${retentionDays}d)`);
         return result.count;
       }
       ```
