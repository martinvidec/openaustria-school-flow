```typescript
import { IsString, IsEnum, IsOptional, IsInt, Min, Max, IsArray, ValidateNested, IsDateString } from 'class-validator';
import { Type } from 'class-transformer';
import { ApiProperty, ApiPropertyOptional } from '@nestjs/swagger';

export enum AttendanceStatusEnum {
  PRESENT = 'PRESENT',
  ABSENT = 'ABSENT',
  LATE = 'LATE',
  EXCUSED = 'EXCUSED',
}

export class AttendanceRecordItemDto {
  @ApiProperty()
  @IsString()
  studentId!: string;

  @ApiProperty({ enum: AttendanceStatusEnum })
  @IsEnum(AttendanceStatusEnum)
  status!: AttendanceStatusEnum;

  @ApiPropertyOptional()
  @IsOptional()
  @IsInt()
  @Min(1)
  @Max(120)
  lateMinutes?: number;
}

export class BulkAttendanceDto {
  @ApiProperty({ type: [AttendanceRecordItemDto] })
  @IsArray()
  @ValidateNested({ each: true })
  @Type(() => AttendanceRecordItemDto)
  records!: AttendanceRecordItemDto[];
}

export class CreateClassBookEntryDto {
  @ApiProperty({ description: 'ClassSubject ID' })
  @IsString()
  classSubjectId!: string;

  @ApiProperty({ description: 'Actual calendar date (ISO)' })
  @IsDateString()
  date!: string;

  @ApiProperty({ description: 'Day of week', enum: ['MONDAY','TUESDAY','WEDNESDAY','THURSDAY','FRIDAY','SATURDAY'] })
  @IsString()
  dayOfWeek!: string;

  @ApiProperty({ description: 'Period number (1-based)' })
  @IsInt()
  @Min(1)
  @Max(20)
  periodNumber!: number;

  @ApiPropertyOptional({ description: 'Week type (BOTH, A, B)', default: 'BOTH' })
  @IsOptional()
  @IsString()
  weekType?: string;
}
```

**2. Create `apps/api/src/modules/classbook/attendance.service.ts`:**

Service that handles:
- `getOrCreateEntry(schoolId, teacherId, dto: CreateClassBookEntryDto)`: Upsert ClassBookEntry by composite key (classSubjectId + date + periodNumber + weekType). Return the entry.
- `getOrCreateEntryByTimetableLesson(schoolId: string, teacherId: string, timetableLessonId: string, date?: string)`: **NEW -- Resolves a TimetableLesson ID to a ClassBookEntry.** Steps: (1) Query `prisma.timetableLesson.findUniqueOrThrow({ where: { id: timetableLessonId }, include: { classSubject: { include: { subject: true, schoolClass: true } } } })`. (2) If no `date` query param provided, calculate today's date. Validate that the TimetableLesson's dayOfWeek matches the date's day of week (or allow override with date param for viewing past entries). (3) Call `getOrCreateEntry()` with the resolved classSubjectId, date, periodNumber, weekType. (4) Return the ClassBookEntryDto with joined fields: subjectName from `classSubject.subject.name`, className from `classSubject.schoolClass.name`, teacherName from Teacher -> Person lookup.
- `getAttendanceForEntry(classBookEntryId: string)`: Fetch all AttendanceRecords for an entry, joined with Student -> Person (firstName, lastName). Sort by lastName, firstName.
- `bulkUpdateAttendance(classBookEntryId: string, teacherKeycloakId: string, dto: BulkAttendanceDto)`: For each record in dto.records, upsert AttendanceRecord by (classBookEntryId + studentId). Use Prisma `upsert` in a transaction. Validate: if status !== 'LATE', set lateMinutes to null (reject lateMinutes on non-LATE). Use `updatedAt` for optimistic locking -- include `updatedAt` check in the where clause and throw ConflictException (409) if stale.
- `setAllPresent(classBookEntryId: string, teacherKeycloakId: string)`: Update all AttendanceRecords for the entry to PRESENT with lateMinutes null.

Inject PrismaService. Use `this.prisma.attendanceRecord.upsert(...)` pattern.

For the join query to get student names:
```typescript
const records = await this.prisma.attendanceRecord.findMany({
  where: { classBookEntryId },
  include: { /* no relation in schema -- query separately */ },
  orderBy: { studentId: 'asc' },
});
// Get student names separately via studentIds
const studentIds = records.map(r => r.studentId);
const students = await this.prisma.student.findMany({
  where: { id: { in: studentIds } },
  include: { person: { select: { firstName: true, lastName: true } } },
});
```

Note: AttendanceRecord doesn't have a direct Prisma relation to Student (by design -- studentId is a string ref). Query students separately and join in-memory.

Controller with route `@Controller('schools/:schoolId/classbook')`. Add to existing attendance controller's base path or create a separate controller that handles the ClassBookEntry-level endpoints:
- `GET /:entryId` -- Get a ClassBookEntry by ID. Returns full entry with content fields.
- `PATCH /:entryId/content` -- Update lesson content (auto-save from frontend). Body: UpdateLessonContentDto. Returns updated ClassBookEntryDto.
- `GET /:entryId/recent` -- Get recent entries for the same classSubject (last 3). Returns ClassBookEntryDto[] (with content fields only, no attendance).
- `GET /by-lesson` -- Get or create entry by lesson params. Query params: classSubjectId, date, periodNumber, weekType. This is the create-on-navigate endpoint -- the frontend calls this when navigating to a lesson, and it returns an existing entry or creates a new one.
- `GET /by-timetable-lesson/:timetableLessonId` -- **CRITICAL: Resolves a TimetableLesson ID to a ClassBookEntry.** This is the primary frontend entry point when a teacher clicks a timetable cell (D-03). Accepts optional `?date=YYYY-MM-DD` query param (defaults to today). Calls `attendanceService.getOrCreateEntryByTimetableLesson(schoolId, teacherId, timetableLessonId, date)`. Returns ClassBookEntryDto with joined subjectName, className, teacherName fields. If the TimetableLesson ID does not exist, returns 404 with message "Stunde nicht gefunden". This endpoint eliminates the need for the frontend to decompose a TimetableLesson into its composite key.

Use `@ApiTags('classbook')` and `@ApiBearerAuth()`.
