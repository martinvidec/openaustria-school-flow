### ab-week.util.ts (resolves Open Question 3 from 06-RESEARCH.md)

```typescript
import { getISOWeek } from 'date-fns';

export type WeekType = 'A' | 'B' | 'BOTH';

/**
 * Returns true if a lesson with weekType=lessonWeekType applies on the given date.
 * Phase 3 convention: ISO-week parity resolves A/B (odd=A, even=B). BOTH always applies.
 * If abWeekEnabled=false for the school, BOTH is the only valid weekType.
 */
export function isWeekCompatible(date: Date, lessonWeekType: string, abWeekEnabled: boolean): boolean {
  if (!abWeekEnabled || lessonWeekType === 'BOTH') return true;
  const isoWeek = getISOWeek(date);
  const targetWeekType: WeekType = isoWeek % 2 === 1 ? 'A' : 'B';
  return lessonWeekType === targetWeekType;
}

export function resolveWeekType(date: Date, abWeekEnabled: boolean): WeekType {
  if (!abWeekEnabled) return 'BOTH';
  const isoWeek = getISOWeek(date);
  return isoWeek % 2 === 1 ? 'A' : 'B';
}

@Controller('absences')   // NON-prefixed — see CLAUDE.md historical note, Nest has global prefix 'api/v1'
@ApiTags('Substitution')
@ApiBearerAuth()
export class TeacherAbsenceController {
  constructor(private readonly service: TeacherAbsenceService) {}

  @Post()
  @CheckPermissions({ action: 'manage', subject: 'substitution' })
  async create(@Body() dto: CreateTeacherAbsenceDto, @Req() req: any) {
    return this.service.create({
      ...dto,
      dateFrom: new Date(dto.dateFrom),
      dateTo: new Date(dto.dateTo),
      schoolId: req.user.schoolId,       // resolved by user-context
      createdBy: req.user.id,             // keycloakUserId
    });
  }

  @Get()
  @CheckPermissions({ action: 'read', subject: 'substitution' })
  async list(@Query() query: ListAbsencesQueryDto, @Req() req: any) {
    return this.service.findManyForSchool(req.user.schoolId, query);
  }

  @Get(':id')
  @CheckPermissions({ action: 'read', subject: 'substitution' })
  async getOne(@Param('id') id: string) {
    return this.service.findOne(id);
  }

  @Delete(':id')
  @CheckPermissions({ action: 'manage', subject: 'substitution' })
  @HttpCode(204)
  async cancel(@Param('id') id: string, @Req() req: any) {
    await this.service.cancel(id, req.user.id);
  }
}
```

```typescript
@Injectable()
export class SubstitutionService {
  constructor(private readonly prisma: PrismaService) {}

  async assignSubstitute(input: AssignSubstituteInput): Promise<SubstitutionDto> {
    return this.prisma.$transaction(async (tx) => {
      // Re-fetch substitution with FOR UPDATE semantics
      const sub = await tx.substitution.findUniqueOrThrow({
        where: { id: input.substitutionId },
        include: { absence: true },
      });

      if (sub.status !== 'PENDING' && sub.status !== 'DECLINED') {
        throw new ConflictException('Substitution cannot be reassigned in its current state');
      }

      // Pitfall 2: re-run hard filter for the selected candidate inside the transaction
      // (RankingService.passesHardFilters is in Plan 03 — for now inline the minimal check)
      const conflictingLesson = await tx.timetableLesson.findFirst({
        where: {
          teacherId: input.candidateTeacherId,
          dayOfWeek: sub.dayOfWeek,
          periodNumber: sub.periodNumber,
          runId: { in: await this.getActiveRunIds(tx, sub) },  // helper
        },
      });

      const conflictingSubstitution = await tx.substitution.findFirst({
        where: {
          substituteTeacherId: input.candidateTeacherId,
          date: sub.date,
          periodNumber: sub.periodNumber,
          status: { in: ['OFFERED', 'CONFIRMED'] },
          id: { not: sub.id },
        },
      });

      if (conflictingLesson || conflictingSubstitution) {
        throw new ConflictException('Vertretung kann nicht vergeben werden: Lehrer/in ist nicht mehr verfügbar.');
      }

      const updated = await tx.substitution.update({
        where: { id: sub.id },
        data: {
          substituteTeacherId: input.candidateTeacherId,
          type: 'SUBSTITUTED',
          status: 'OFFERED',
          offeredAt: new Date(),
        },
        include: this.includeForDto(),
      });

      return this.toDto(updated);
    }, { isolationLevel: 'Serializable' });
  }

  async respondToOffer(input: RespondToOfferInput): Promise<SubstitutionDto> {
    return this.prisma.$transaction(async (tx) => {
      const sub = await tx.substitution.findUniqueOrThrow({
        where: { id: input.substitutionId },
        include: { absence: true },
      });

      if (sub.status !== 'OFFERED') {
        throw new ConflictException('Substitution is not in OFFERED state');
      }

      // Verify userId matches the assigned substitute via Person.keycloakUserId
      const substitute = await tx.teacher.findUniqueOrThrow({
        where: { id: sub.substituteTeacherId! },
        include: { person: { select: { keycloakUserId: true } } },
      });

      if (substitute.person.keycloakUserId !== input.userId) {
        throw new ForbiddenException('Nur der zugewiesene Vertretungslehrer kann antworten');
      }

      const newStatus = input.accept ? 'CONFIRMED' : 'DECLINED';
      const updated = await tx.substitution.update({
        where: { id: sub.id },
        data: {
          status: newStatus,
          respondedAt: new Date(),
        },
        include: this.includeForDto(),
      });

      // D-14: on CONFIRMED accept, create ClassBookEntry with substitute teacher
      if (input.accept && updated.type === 'SUBSTITUTED') {
        await this.ensureClassBookEntryForSubstitution(tx, updated);
      }

      return this.toDto(updated);
    });
  }

  async setEntfall(input: SetEntfallInput): Promise<SubstitutionDto> {
    return this.prisma.$transaction(async (tx) => {
      const sub = await tx.substitution.findUniqueOrThrow({ where: { id: input.substitutionId } });
      if (sub.status === 'CONFIRMED' && sub.type === 'ENTFALL') return this.findOne(sub.id);  // idempotent

      const updated = await tx.substitution.update({
        where: { id: sub.id },
        data: { type: 'ENTFALL', status: 'CONFIRMED', respondedAt: new Date() },
        include: this.includeForDto(),
      });

      // D-14: ENTFALL => no ClassBookEntry. If one exists from a previous transition, delete it.
      await tx.classBookEntry.deleteMany({ where: { substitutionId: sub.id } });

      return this.toDto(updated);
    });
  }

  async setStillarbeit(input: SetStillarbeitInput): Promise<SubstitutionDto> {
    return this.prisma.$transaction(async (tx) => {
      const sub = await tx.substitution.findUniqueOrThrow({ where: { id: input.substitutionId } });

      const updated = await tx.substitution.update({
        where: { id: sub.id },
        data: {
          type: 'STILLARBEIT',
          status: 'CONFIRMED',
          substituteTeacherId: input.supervisorTeacherId ?? null,
          respondedAt: new Date(),
        },
        include: this.includeForDto(),
      });

      // D-14: create ClassBookEntry with "Stillarbeit" marker
      await this.ensureStillarbeitClassBookEntry(tx, updated, input.supervisorTeacherId);

      return this.toDto(updated);
    });
  }

  async findManyPending(schoolId: string): Promise<SubstitutionDto[]> {
    const rows = await this.prisma.substitution.findMany({
      where: {
        status: { in: ['PENDING', 'OFFERED', 'DECLINED'] },
        absence: { schoolId },
      },
      include: this.includeForDto(),
      orderBy: [{ date: 'asc' }, { periodNumber: 'asc' }],
    });
    return rows.map((r) => this.toDto(r));
  }

  async findOne(id: string): Promise<SubstitutionDto> {
    const row = await this.prisma.substitution.findUniqueOrThrow({
      where: { id },
      include: this.includeForDto(),
    });
    return this.toDto(row);
  }

  // --- private helpers ---

  private async ensureClassBookEntryForSubstitution(tx: any, sub: any) {
    // Per D-14: teacherId = substitute.id, substitutionId FK set. Upsert by substitutionId unique.
    await tx.classBookEntry.upsert({
      where: { substitutionId: sub.id },
      create: {
        classSubjectId: sub.classSubjectId,
        date: sub.date,
        periodNumber: sub.periodNumber,
        teacherId: sub.substituteTeacherId,
        substitutionId: sub.id,
        // thema/lehrstoff left null — substitute fills in during the lesson
      },
      update: { teacherId: sub.substituteTeacherId },
    });
  }

  private async ensureStillarbeitClassBookEntry(tx: any, sub: any, supervisorTeacherId?: string) {
    // Per D-14 + Open Question 4 resolution: thema="Stillarbeit", lehrstoff=null
    await tx.classBookEntry.upsert({
      where: { substitutionId: sub.id },
      create: {
        classSubjectId: sub.classSubjectId,
        date: sub.date,
        periodNumber: sub.periodNumber,
        teacherId: supervisorTeacherId ?? sub.originalTeacherId,  // fallback to original to satisfy FK
        substitutionId: sub.id,
        thema: 'Stillarbeit',
        lehrstoff: null,
      },
      update: {
        teacherId: supervisorTeacherId ?? sub.originalTeacherId,
        thema: 'Stillarbeit',
      },
    });
  }

  private includeForDto() {
    return {
      absence: { include: { teacher: { include: { person: true } } } },
      // lesson is a soft ref — join via denormalized fields would live here in a future refactor
    };
  }

  private toDto(row: any): SubstitutionDto {
    // Map Prisma row to shared DTO shape. See packages/shared/src/types/substitution.ts
    return {
      id: row.id,
      absenceId: row.absenceId,
      lessonId: row.lessonId,
      classSubjectId: row.classSubjectId,
      dayOfWeek: row.dayOfWeek,
      periodNumber: row.periodNumber,
      weekType: row.weekType,
      date: row.date.toISOString(),
      type: row.type,
      status: row.status,
      originalTeacherId: row.originalTeacherId,
      originalTeacherName: `${row.absence?.teacher?.person?.firstName ?? ''} ${row.absence?.teacher?.person?.lastName ?? ''}`.trim(),
      substituteTeacherId: row.substituteTeacherId,
      substituteTeacherName: null,  // filled by controller via join if needed
      substituteRoomId: row.substituteRoomId,
      offeredAt: row.offeredAt?.toISOString() ?? null,
      respondedAt: row.respondedAt?.toISOString() ?? null,
      createdAt: row.createdAt.toISOString(),
      updatedAt: row.updatedAt.toISOString(),
      subjectAbbreviation: '',  // filled in Plan 04 view-layer
      subjectName: '',
      className: '',
    };
  }

  private async getActiveRunIds(tx: any, sub: any): Promise<string[]> {
    // Helper: return [activeRunId] for the school this substitution belongs to.
    // Lookup via absence.schoolId
    const absence = await tx.teacherAbsence.findUniqueOrThrow({ where: { id: sub.absenceId } });
    const runs = await tx.timetableRun.findMany({ where: { schoolId: absence.schoolId, isActive: true }, select: { id: true } });
    return runs.map((r: any) => r.id);
  }
}
```

Notes on the implementation choices:
- `includeForDto()` helper centralizes the Prisma join graph — keep it private so all read paths return the same shape.
- `toDto()` leaves some fields empty (`subjectAbbreviation`, `subjectName`, `className`, `substituteTeacherName`) — these are filled by Plan 04 which rewrites the view layer. Plan 02's job is the lifecycle correctness, not the read-shape completeness.
- ClassBookEntry upsert uses `substitutionId` as the unique key (from Plan 01 schema).
- For Stillarbeit without a supervisor, fallback to `originalTeacherId` for the ClassBookEntry.teacherId FK (we cannot leave it null if the column is required — check schema). If ClassBookEntry.teacherId is nullable in schema, set to `supervisorTeacherId ?? null` instead. Check schema.prisma during implementation.
- The `tx: any` typing is a pragmatic shortcut — the real Prisma TransactionClient type is `Prisma.TransactionClient`. Use it if type imports are already in the file.

@Controller('substitutions')   // NON-prefixed per CLAUDE.md
@ApiTags('Substitution')
@ApiBearerAuth()
export class SubstitutionController {
  constructor(private readonly service: SubstitutionService) {}

  @Get()
  @CheckPermissions({ action: 'read', subject: 'substitution' })
  async list(@Req() req: any) {
    return this.service.findManyPending(req.user.schoolId);
  }

  @Get(':id')
  @CheckPermissions({ action: 'read', subject: 'substitution' })
  async getOne(@Param('id') id: string) {
    return this.service.findOne(id);
  }

  @Post(':id/assign')
  @CheckPermissions({ action: 'manage', subject: 'substitution' })
  async assign(@Param('id') id: string, @Body() dto: AssignSubstituteDto, @Req() req: any) {
    return this.service.assignSubstitute({
      substitutionId: id,
      candidateTeacherId: dto.candidateTeacherId,
      assignedBy: req.user.id,
    });
  }

  @Patch(':id/respond')
  @CheckPermissions({ action: 'update', subject: 'substitution' })
  async respond(@Param('id') id: string, @Body() dto: RespondToOfferDto, @Req() req: any) {
    return this.service.respondToOffer({
      substitutionId: id,
      userId: req.user.id,
      accept: dto.accept,
      declineReason: dto.declineReason,
    });
  }

  @Post(':id/entfall')
  @CheckPermissions({ action: 'manage', subject: 'substitution' })
  async entfall(@Param('id') id: string, @Req() req: any) {
    return this.service.setEntfall({ substitutionId: id, actorUserId: req.user.id });
  }

  @Post(':id/stillarbeit')
  @CheckPermissions({ action: 'manage', subject: 'substitution' })
  async stillarbeit(@Param('id') id: string, @Body() dto: SetStillarbeitDto, @Req() req: any) {
    return this.service.setStillarbeit({
      substitutionId: id,
      supervisorTeacherId: dto.supervisorTeacherId,
      actorUserId: req.user.id,
    });
  }
}
