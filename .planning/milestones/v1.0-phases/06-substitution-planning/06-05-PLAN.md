**Step A — Create `apps/web/src/hooks/useAbsences.ts`:**

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { apiFetch } from '@/lib/api';
import type { TeacherAbsenceDto } from '@schoolflow/shared';

export const absenceKeys = {
  all: ['absences'] as const,
  list: (status?: string) => ['absences', 'list', { status }] as const,
  detail: (id: string) => ['absences', 'detail', id] as const,
};

export interface CreateTeacherAbsencePayload {
  teacherId: string;
  dateFrom: string;
  dateTo: string;
  periodFrom?: number;
  periodTo?: number;
  reason: 'KRANK' | 'FORTBILDUNG' | 'DIENSTREISE' | 'SCHULVERANSTALTUNG' | 'ARZTTERMIN' | 'SONSTIGES';
  note?: string;
}

export function useAbsences(opts: { status?: 'ACTIVE' | 'CANCELLED' | 'COMPLETED' } = {}) {
  return useQuery({
    queryKey: absenceKeys.list(opts.status),
    queryFn: async () => {
      const qs = opts.status ? `?status=${opts.status}` : '';
      return apiFetch<TeacherAbsenceDto[]>(`/absences${qs}`);
    },
  });
}

export function useCreateAbsence() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async (payload: CreateTeacherAbsencePayload) => {
      return apiFetch<{ absence: TeacherAbsenceDto; affectedLessonCount: number }>('/absences', {
        method: 'POST',
        body: payload,
      });
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: absenceKeys.all });
      queryClient.invalidateQueries({ queryKey: ['substitutions', 'pending'] });
    },
  });
}

export function useCancelAbsence() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async (id: string) => {
      return apiFetch<void>(`/absences/${id}`, { method: 'DELETE' });
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: absenceKeys.all });
      queryClient.invalidateQueries({ queryKey: ['substitutions', 'pending'] });
    },
  });
}
```

**Step B — Create `apps/web/src/hooks/useSubstitutions.ts`:**

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { apiFetch } from '@/lib/api';
import type { SubstitutionDto } from '@schoolflow/shared';

export const substitutionKeys = {
  all: ['substitutions'] as const,
  pending: ['substitutions', 'pending'] as const,
  detail: (id: string) => ['substitutions', 'detail', id] as const,
};

export function usePendingSubstitutions() {
  return useQuery({
    queryKey: substitutionKeys.pending,
    queryFn: () => apiFetch<SubstitutionDto[]>('/substitutions'),
  });
}

export function useAssignSubstitute() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async ({ substitutionId, candidateTeacherId }: { substitutionId: string; candidateTeacherId: string }) => {
      return apiFetch<SubstitutionDto>(`/substitutions/${substitutionId}/assign`, {
        method: 'POST',
        body: { candidateTeacherId },
      });
    },
    onSuccess: (_, vars) => {
      queryClient.invalidateQueries({ queryKey: substitutionKeys.all });
      queryClient.invalidateQueries({ queryKey: ['ranked-candidates', vars.substitutionId] });
    },
  });
}

export function useSetEntfall() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async (substitutionId: string) => {
      return apiFetch<SubstitutionDto>(`/substitutions/${substitutionId}/entfall`, { method: 'POST' });
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: substitutionKeys.all });
    },
  });
}

export function useSetStillarbeit() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: async ({ substitutionId, supervisorTeacherId }: { substitutionId: string; supervisorTeacherId?: string }) => {
      return apiFetch<SubstitutionDto>(`/substitutions/${substitutionId}/stillarbeit`, {
        method: 'POST',
        body: { supervisorTeacherId },
      });
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: substitutionKeys.all });
    },
  });
}
```

```typescript
import { useQuery } from '@tanstack/react-query';
import { apiFetch } from '@/lib/api';
import type { FairnessStatRow } from '@schoolflow/shared';

export type StatsWindow = 'week' | 'month' | 'semester' | 'schoolYear' | 'custom';

export function useSubstitutionStats(window: StatsWindow, customStart?: string, customEnd?: string) {
  const params = new URLSearchParams({ window });
  if (customStart) params.set('customStart', customStart);
  if (customEnd) params.set('customEnd', customEnd);

  return useQuery({
    queryKey: ['substitution-stats', window, customStart, customEnd],
    queryFn: () => apiFetch<FairnessStatRow[]>(`/substitution-stats?${params}`),
  });
}
```

```tsx
import { Button } from '@/components/ui/button';
import { Card } from '@/components/ui/card';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';

interface Props {

  onSubmit: (payload: CreateTeacherAbsencePayload) => Promise<{ affectedLessonCount: number }>;
}

const REASON_OPTIONS: Array<{ value: CreateTeacherAbsencePayload['reason']; label: string }> = [

  { value: 'FORTBILDUNG', label: 'Fortbildung' },
  { value: 'DIENSTREISE', label: 'Dienstreise' },
  { value: 'SCHULVERANSTALTUNG', label: 'Schulveranstaltung' },
  { value: 'ARZTTERMIN', label: 'Arzttermin' },
  { value: 'SONSTIGES', label: 'Sonstiges' },
];

export function AbsenceForm({ teachers, onSubmit }: Props) {
  const today = new Date().toISOString().slice(0, 10);
  const [teacherId, setTeacherId] = useState('');
  const [dateFrom, setDateFrom] = useState(today);
  const [dateTo, setDateTo] = useState(today);
  const [periodFrom, setPeriodFrom] = useState<string>('');
  const [periodTo, setPeriodTo] = useState<string>('');
  const [reason, setReason] = useState<CreateTeacherAbsencePayload['reason']>('KRANK');
  const [note, setNote] = useState('');
  const [submitting, setSubmitting] = useState(false);
  const [dateError, setDateError] = useState<string | null>(null);

  const validate = (): boolean => {
    if (!teacherId) { toast.error('Bitte waehlen Sie eine Lehrkraft aus.'); return false; }
    if (new Date(dateTo) < new Date(dateFrom)) {
      setDateError('Das Enddatum muss nach dem Startdatum liegen.');
      return false;
    }
    setDateError(null);
    if (periodFrom && periodTo && parseInt(periodTo) < parseInt(periodFrom)) {
      toast.error('Die End-Stunde muss nach der Anfangs-Stunde liegen.');
      return false;
    }
    return true;
  };

    e.preventDefault();
    if (!validate()) return;
    setSubmitting(true);
    try {
      const result = await onSubmit({
        teacherId,
        dateFrom,
        dateTo,
        periodFrom: periodFrom ? parseInt(periodFrom) : undefined,
        periodTo: periodTo ? parseInt(periodTo) : undefined,
        reason,
        note: note || undefined,
      });
      if (result.affectedLessonCount === 0) {
        toast.warning('Keine Unterrichtsstunden im angegebenen Zeitraum gefunden.');
      } else {
        toast.success(`Abwesenheit erfasst. ${result.affectedLessonCount} Stunden betroffen.`);
      }
      // Reset form
      setTeacherId('');
      setNote('');
    } catch (err) {
      toast.error('Abwesenheiten konnten nicht erfasst werden. Bitte versuchen Sie es spaeter erneut.');
    } finally {
      setSubmitting(false);
    }
  };

  return (
    <Card className="p-6">
      <form onSubmit={handleSubmit} className="space-y-4">
        <div>
          <Label htmlFor="teacherId">Lehrer/in</Label>
          <Select value={teacherId} onValueChange={setTeacherId}>
            <SelectTrigger id="teacherId">
              <SelectValue placeholder="Lehrkraft auswaehlen" />
            </SelectTrigger>
            <SelectContent>
              {teachers.map((t) => (
                <SelectItem key={t.id} value={t.id}>{t.lastName}, {t.firstName}</SelectItem>
              ))}
            </SelectContent>
          </Select>
        </div>

        <div className="grid grid-cols-2 gap-4">
          <div>
            <Label htmlFor="dateFrom">Von</Label>
            <Input id="dateFrom" type="date" value={dateFrom} onChange={(e) => setDateFrom(e.target.value)} required />
          </div>
          <div>
            <Label htmlFor="dateTo">Bis</Label>
            <Input id="dateTo" type="date" value={dateTo} onChange={(e) => setDateTo(e.target.value)} required />
          </div>
        </div>
        {dateError && <p className="text-destructive text-sm" role="alert">{dateError}</p>}

        <div className="grid grid-cols-2 gap-4">
          <div>
            <Label htmlFor="periodFrom">Von Stunde (optional)</Label>
```
