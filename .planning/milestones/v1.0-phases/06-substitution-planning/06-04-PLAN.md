**Step D — GREEN phase: Rewrite `timetable.service.ts::getView()` to be overlay-aware.**

Current getView() fetches TimetableLessons and returns them with scalar `changeType`/`originalTeacherSurname`/`originalRoomName`. The rewrite:

1. Accept optional `query.date?: string`. Parse into Date.
2. If `date` is undefined, behavior is unchanged — return the recurring plan exactly as before. This is backward-compatible per 06-RESEARCH.md Option A recommendation.
3. If `date` is provided:
   a. Compute the week of that date (`startOfWeek(date, { weekStartsOn: 1 })` and `endOfWeek(date, { weekStartsOn: 1 })` from date-fns — Monday to Sunday).
   b. After fetching the TimetableLesson rows, fetch all Substitution rows overlapping the week:
      ```typescript
      const overlays = await this.prisma.substitution.findMany({
        where: {
          lessonId: { in: lessonIds },
          date: { gte: weekStart, lte: weekEnd },
          status: { in: ['CONFIRMED', 'OFFERED'] },
        },
        include: {
          substituteTeacher: { include: { person: true } },
          substituteRoom: true,
          absence: { include: { teacher: { include: { person: true } } } },
        },
      });
      ```
   c. Build a Map keyed by `${lessonId}:${isoDate}` -> overlay for fast lookup.
   d. When mapping each TimetableLesson to TimetableLessonViewDto, iterate the dates in the week (5-7 depending on school days). For each (lesson, date) pair:
      - Compute the matching weekType using `isWeekCompatible` from the ab-week.util.ts created in Plan 02 — skip if lesson.weekType is not compatible with date's ISO week.
      - Lookup `overlays.get(`${lesson.id}:${date.toISOString().slice(0,10)}`)`.
      - If overlay found:
        * Apply per-type:
          - SUBSTITUTED: `changeType='substitution'`, `originalTeacherSurname=overlay.absence.teacher.person.lastName`, `teacherId=overlay.substituteTeacherId`, `teacherSurname=overlay.substituteTeacher.person.lastName`, `teacherFirstName=overlay.substituteTeacher.person.firstName`. If `substituteRoomId` present: also override `roomId`/`roomName`/`originalRoomName`.
          - ENTFALL: `changeType='cancelled'`, preserve teacher/room fields (rendered strikethrough by frontend).
          - STILLARBEIT: `changeType='stillarbeit'`, set `teacherId=overlay.substituteTeacherId` (supervisor, may be null), `originalTeacherSurname=overlay.absence.teacher.person.lastName`.
      - If no overlay found: preserve existing scalar columns if set (legacy Phase 4 edits) OR leave changeType null for an unmodified recurring lesson.
4. Return the augmented DTO array.

IMPORTANT: The existing `getView()` must remain backward-compatible for `query.date=undefined`. Do NOT break Phase 4 view callers.

```typescript
import { Injectable } from '@nestjs/common';
import { startOfWeek, endOfWeek, startOfMonth, endOfMonth, startOfYear, endOfYear } from 'date-fns';
import { PrismaService } from '../../../config/database/prisma.service';
import { StatisticsService } from '../../classbook/statistics.service';  // For getSemesterDateRange reuse
import { calculateWerteinheiten } from '../../teacher/werteinheiten.util';

export type StatsWindow = 'week' | 'month' | 'semester' | 'schoolYear' | 'custom';

export interface StatsWindowInput {
  window: StatsWindow;
  customStart?: Date;
  customEnd?: Date;
}

export interface FairnessStatRow {
  teacherId: string;
  teacherName: string;
  givenCount: number;
  givenWerteinheiten: number;
  receivedCount: number;
  entfallAffectedCount: number;
  stillarbeitAffectedCount: number;
  deltaVsAverage: number;
}

@Injectable()
export class SubstitutionStatsService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly classbookStats: StatisticsService,  // reuse Austrian semester logic
  ) {}

  async getFairnessStats(schoolId: string, windowInput: StatsWindowInput): Promise<FairnessStatRow[]> {
    const { start, end } = this.resolveWindow(windowInput);

    // Fetch all teachers in school
    const teachers = await this.prisma.teacher.findMany({
      where: { schoolId },
      include: { person: true },
    });

    // Fetch all Substitution rows in window for the school (joined through absence.schoolId)
    const subs = await this.prisma.substitution.findMany({
      where: {
        date: { gte: start, lte: end },
        status: 'CONFIRMED',
        absence: { schoolId },
      },
      include: {
        // Need to compute werteinheiten — fetch related lesson metadata (via denormalized fields)
      },
    });

    // Group by substituteTeacherId for "given" and originalTeacherId for "received/affected"
    const givenByTeacher = new Map<string, { count: number; werteinheiten: number }>();
    const receivedByTeacher = new Map<string, number>();
    const entfallAffectedByTeacher = new Map<string, number>();
    const stillarbeitAffectedByTeacher = new Map<string, number>();

    for (const sub of subs) {
      // Given (only SUBSTITUTED type counts as "given" by a teacher; STILLARBEIT supervisor also counts if set)
      if ((sub.type === 'SUBSTITUTED' || sub.type === 'STILLARBEIT') && sub.substituteTeacherId) {
        const we = this.calculateSubstitutionWerteinheiten(sub);
        const current = givenByTeacher.get(sub.substituteTeacherId) ?? { count: 0, werteinheiten: 0 };
        givenByTeacher.set(sub.substituteTeacherId, {
          count: current.count + 1,
          werteinheiten: current.werteinheiten + we,
        });
      }

      // Received / affected (always by originalTeacherId)
      receivedByTeacher.set(sub.originalTeacherId, (receivedByTeacher.get(sub.originalTeacherId) ?? 0) + 1);

      if (sub.type === 'ENTFALL') {
        entfallAffectedByTeacher.set(sub.originalTeacherId, (entfallAffectedByTeacher.get(sub.originalTeacherId) ?? 0) + 1);
      } else if (sub.type === 'STILLARBEIT') {
        stillarbeitAffectedByTeacher.set(sub.originalTeacherId, (stillarbeitAffectedByTeacher.get(sub.originalTeacherId) ?? 0) + 1);
      }
    }

    // Compute school average for fairness delta (D-17)
    const allGivenCounts = Array.from(givenByTeacher.values()).map((v) => v.count);
    const schoolAverage = teachers.length > 0
      ? allGivenCounts.reduce((a, b) => a + b, 0) / teachers.length
      : 0;

    return teachers.map((t) => {
      const given = givenByTeacher.get(t.id) ?? { count: 0, werteinheiten: 0 };
      const received = receivedByTeacher.get(t.id) ?? 0;
      const entfall = entfallAffectedByTeacher.get(t.id) ?? 0;
      const stillarbeit = stillarbeitAffectedByTeacher.get(t.id) ?? 0;
      return {
        teacherId: t.id,
        teacherName: `${t.person.lastName}, ${t.person.firstName}`,
        givenCount: given.count,
        givenWerteinheiten: given.werteinheiten,
        receivedCount: received,
        entfallAffectedCount: entfall,
        stillarbeitAffectedCount: stillarbeit,
        deltaVsAverage: given.count - schoolAverage,
      };
    }).sort((a, b) => b.deltaVsAverage - a.deltaVsAverage);
  }

  private calculateSubstitutionWerteinheiten(sub: any): number {
    // Per D-17: weight given hours by Werteinheiten so a Doppelstunde counts for 2 periods.
    // Phase 2 werteinheiten.util.ts operates on (Lehrverpflichtungsgruppe, periodCount).
    // For v1, treat each substitution as 1 period with default factor. Doppelstunde handling
    // requires looking up the underlying lesson, which is feasible via denormalized classSubjectId.
    // Simple v1: return 1.0 per substitution. Annotate as TODO for future refinement.
    // If schema has a denormalized lesson weekType+periodNumber spanning 2 periods, sum those.
    return 1.0;
  }

  private resolveWindow(input: StatsWindowInput): { start: Date; end: Date } {
    const now = new Date();
    switch (input.window) {
      case 'week':
        return { start: startOfWeek(now, { weekStartsOn: 1 }), end: endOfWeek(now, { weekStartsOn: 1 }) };
      case 'month':
        return { start: startOfMonth(now), end: endOfMonth(now) };
      case 'semester': {
        // Reuse Phase 5 semester range utility — call StatisticsService.getSemesterDateRange
        // If the method is private, grep confirms this and we either make it public OR duplicate the logic here.
        // Austrian semester: Sep 1 - Jan 31 (Wintersemester), Feb 1 - Jun 30 (Sommersemester)
        const month = now.getMonth();  // 0=Jan
        if (month >= 8) {
          // Wintersemester (Sep-Jan of next year)
          return { start: new Date(now.getFullYear(), 8, 1), end: new Date(now.getFullYear() + 1, 0, 31) };
        } else if (month <= 0) {
          // Wintersemester continuing from previous year
          return { start: new Date(now.getFullYear() - 1, 8, 1), end: new Date(now.getFullYear(), 0, 31) };
        } else {
          // Sommersemester (Feb-Jun)
          return { start: new Date(now.getFullYear(), 1, 1), end: new Date(now.getFullYear(), 5, 30) };
        }
      }
      case 'schoolYear': {
        // Austrian school year: Sep 1 - Jun 30
        const month = now.getMonth();
        const year = month >= 8 ? now.getFullYear() : now.getFullYear() - 1;
        return { start: new Date(year, 8, 1), end: new Date(year + 1, 5, 30) };
      }
      case 'custom':
        if (!input.customStart || !input.customEnd) {
          throw new Error('customStart and customEnd required for custom window');
        }
        return { start: input.customStart, end: input.customEnd };
    }
  }
}
```
