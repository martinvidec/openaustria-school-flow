    ```typescript
    @Get('ready')
    @Public()
    @ApiOperation({ summary: 'Readiness check - verifies external dependencies' })
    @ApiResponse({ status: 200, description: 'All dependencies healthy' })
    @ApiResponse({ status: 503, description: 'One or more dependencies unhealthy' })
    async ready(@Res() res: FastifyReply) {
      const checks = { database: false, redis: false, keycloak: false };

      // PostgreSQL
      try {
        await this.prisma.$queryRaw`SELECT 1`;
        checks.database = true;
      } catch {}

      // Redis
      try {
        const pong = await this.redis.ping();
        checks.redis = pong === 'PONG';
      } catch {}

      // Keycloak
      try {
        const response = await fetch(`${this.keycloakUrl}/realms/${this.realm}`);
        checks.keycloak = response.ok;
      } catch {}

      const allHealthy = Object.values(checks).every(Boolean);
      const body = {
        status: allHealthy ? 'ready' : 'degraded',
        checks,
        timestamp: new Date().toISOString(),
      };

      return res.status(allHealthy ? 200 : 503).send(body);
    }
    ```

    Use `@Res()` from `@nestjs/common` with `FastifyReply` for dynamic status code.
    Inject PrismaService, ConfigService via constructor. Create ioredis Redis instance in constructor.
    Get Keycloak URL from `config.get('KEYCLOAK_URL', 'http://localhost:8080')` and realm from `config.get('KEYCLOAK_REALM', 'schoolflow')`.

    **Step 1: Create backup.sh (D-09)**
    At `docker/scripts/backup.sh`:
    ```bash
    #!/bin/bash
    set -euo pipefail
    ```
    Features:
    - `--dry-run` flag that prints what would be done without executing
    - `BACKUP_DIR` env var (default: `/backups`)
    - `RETAIN_DAYS` env var (default: `7`)
    - `COMPOSE_FILE` env var (default: `../docker-compose.yml`)
    - Timestamp-named backups: `postgres_YYYYMMDD_HHMMSS.sql.gz`, `redis_YYYYMMDD_HHMMSS.rdb`
    - Pre-check: verify postgres container is running via `docker compose ps postgres --format '{{.State}}' | grep -q running` (Pitfall 5)
    - PostgreSQL dump: `docker compose exec -T postgres pg_dump -U schoolflow schoolflow | gzip > "$BACKUP_DIR/postgres_$TIMESTAMP.sql.gz"`
    - Redis snapshot: `docker compose exec -T redis redis-cli BGSAVE`, sleep 2, `docker compose cp redis:/data/dump.rdb "$BACKUP_DIR/redis_$TIMESTAMP.rdb"`
    - File uploads: `tar czf "$BACKUP_DIR/uploads_$TIMESTAMP.tar.gz" "$UPLOAD_DIR"` if UPLOAD_DIR is set and exists
    - Retention cleanup: `find "$BACKUP_DIR" -name "*.gz" -mtime +$RETAIN_DAYS -delete` and same for `.rdb`
    - Print summary with file sizes
    - Make executable: `chmod +x`

    **Step 2: Create restore.sh (D-09)**
    At `docker/scripts/restore.sh`:
    ```bash
    #!/bin/bash
    set -euo pipefail
    ```
    Features:
    - `--dry-run` flag
    - `--file <backup_prefix>` flag to specify which backup to restore (timestamp prefix)
    - Pre-check: verify backup files exist
    - PostgreSQL restore: `gunzip -c "$BACKUP_DIR/postgres_$FILE.sql.gz" | docker compose exec -T postgres psql -U schoolflow schoolflow`
    - Redis restore: `docker compose cp "$BACKUP_DIR/redis_$FILE.rdb" redis:/data/dump.rdb`, then restart redis
    - File uploads restore: `tar xzf "$BACKUP_DIR/uploads_$FILE.tar.gz" -C /`
    - **Integrity verification** (D-09): After restore, compare row counts for key tables (schools, persons, teachers, students, notifications) against a manifest file created during backup. Backup.sh writes `$BACKUP_DIR/manifest_$TIMESTAMP.json` with table row counts via `docker compose exec -T postgres psql -U schoolflow schoolflow -t -c "SELECT count(*) FROM schools"`. Restore.sh reads manifest and compares post-restore counts.
    - Make executable: `chmod +x`
